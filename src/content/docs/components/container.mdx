import { Tabs, TabItem } from '@astrojs/starlight/components';

`Container` centers the content and adds horizontal padding based on screen size.

<Tabs>
  <TabItem label='HTML'>
    ```html
    <div class="container mx-auto px-4 sm:px-6 lg:px-8">
      <!-- Content -->
    </div>
    ```
  </TabItem>
  <TabItem label='Vue'>
    ```vue
    <template>
      <div class="container mx-auto px-4 sm:px-6 lg:px-8">
        <!-- Content -->
      </div>
    </template>
    ```
  </TabItem>
</Tabs>
